#define primerLed 2
#define cantidadLeds 15
#define retardo 500

void setup() {
  for (int i = 0; i < cantidadLeds; i++) {
    pinMode(primerLed + i, OUTPUT);
  }
}

void loop() {
  for (int i = 0; i < cantidadLeds; i++) {
    digitalWrite(primerLed + i, HIGH);
    delay(retardo);
    digitalWrite(primerLed + i, LOW);
  }
}
